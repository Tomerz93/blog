---
lang: typescript
name: useToggle
description: A simple hook to toggle visibility in React
---

```ts
import React from 'react';

const useToggle = (isOpen: boolean = false) => {
  const [isVisible, setIsVisible] = React.useState(isOpen);
  const toggle = () => React.useCallBack(() => setIsVisible((p) => !p));
  return {
    toggle,
    isVisible,
  };
};
```

This is a simple hook that you can use in your react Project, I found myself having to toggle off visibility quite often for models etc' so using this hook you save yourslves some time and keep from writing the same logic all over the place.
